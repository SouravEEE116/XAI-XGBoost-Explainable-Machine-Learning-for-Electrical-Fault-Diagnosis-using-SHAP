# STEP 7: FEATURE ENGINEERING

def extract_features(df):
    df_new = df.copy()
    
    # RMS
    df_new['I_rms'] = np.sqrt((df['Ia']**2 + df['Ib']**2 + df['Ic']**2)/3)
    df_new['V_rms'] = np.sqrt((df['Va']**2 + df['Vb']**2 + df['Vc']**2)/3)

    # Mean
    df_new['I_mean'] = df[['Ia','Ib','Ic']].mean(axis=1)
    df_new['V_mean'] = df[['Va','Vb','Vc']].mean(axis=1)

    # Std
    df_new['I_std'] = df[['Ia','Ib','Ic']].std(axis=1)
    df_new['V_std'] = df[['Va','Vb','Vc']].std(axis=1)

    # Max
    df_new['I_max'] = df[['Ia','Ib','Ic']].max(axis=1)
    df_new['V_max'] = df[['Va','Vb','Vc']].max(axis=1)

    # Min
    df_new['I_min'] = df[['Ia','Ib','Ic']].min(axis=1)
    df_new['V_min'] = df[['Va','Vb','Vc']].min(axis=1)

    # Peak to peak
    df_new['I_ptp'] = df_new['I_max'] - df_new['I_min']
    df_new['V_ptp'] = df_new['V_max'] - df_new['V_min']

    # Energy
    df_new['I_energy'] = df['Ia']**2 + df['Ib']**2 + df['Ic']**2
    df_new['V_energy'] = df['Va']**2 + df['Vb']**2 + df['Vc']**2

    return df_new

final_df = extract_features(iforest_df)

# STEP 8: FAULT LABELING

def label_fault_type(row):
    G, C, B, A = row['G'], row['C'], row['B'], row['A']
    if G==0 and C==0 and B==0 and A==0:
        return 'No Fault'
    if G==1 and C==0 and B==0 and A==1:
        return 'LG'
    if G==0 and C==1 and B==1 and A==0:
        return 'LL'
    if G==1 and C==0 and B==1 and A==1:
        return 'LLG'
    if G==0 and C==1 and B==1 and A==1:
        return 'LLL'
    if G==1 and C==1 and B==1 and A==1:
        return 'GLLL'
    return 'Unknown'

final_df['Fault Type'] = final_df.apply(label_fault_type, axis=1)
